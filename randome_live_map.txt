[Unit]
Description=map.com.gunicorn socket

[Install]
WantedBy=sockets.target


************** Create System Service File for Gunicorn ******
Syntax:- sudo nano /etc/systemd/system/your_domain.gunicorn.service
Example:- sudo nano /etc/systemd/system/narayaniorder.com.gunicorn.service

cd  /etc/systemd/system/

sudo nano map.com.gunicorn.service


[Unit]
Description=map.com.gunicorn daemon
Requires=map.com.gunicorn.socket
After=network.target

[Service]
User=root
Group=root
WorkingDirectory=/root/map
ExecStart=/root/map/venv/bin/gunicorn \
          --access-logfile - \
          --workers 3 \
          --bind unix:/run/map.com.gunicorn.sock \
          map.wsgi:application

[Install]
WantedBy=multi-user.target

----------------------------------------------------------------
sudo systemctl start map.com.gunicorn.socket

sudo systemctl start map.com.gunicorn.service


-----------------------------------------------------------------

sudo systemctl enable map.com.gunicorn.socket

sudo systemctl enable map.com.gunicorn.service


-----------------------------------------------------------------------

sudo systemctl status map.com.gunicorn.socket

sudo systemctl status map.com.gunicorn.service

-------------------------------------------

sudo systemctl daemon-reload

****************** Create Virtual Host File ******
Syntax:- sudo nano /etc/nginx/sites-available/your_domain
Example:- sudo nano /etc/nginx/sites-available/crenta.in


cd /etc/nginx/sites-available

sudo nano map.com



server{
    listen 80;
    listen [::]:80;

    server_name 103.150.136.82 map.digitalkarmala.com www.map.digitalkarmala.com;
    client_max_body_size 500M;

    location = /favicon.ico { access_log off; log_not_found off; }

    location / {
        proxy_set_header Host $http_host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_pass http://unix:/run/map.com.gunicorn.sock;
    }    
}
